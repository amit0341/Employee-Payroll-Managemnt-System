CREATE TABLE staffinformation (`Emp_ID` INT NOT NULL AUTO_INCREMENT, `FirstName` VARCHAR(50), `LastName` VARCHAR(50), `DateofBirth` DATE, `Email` VARCHAR(50), `Address1` VARCHAR(50), `Address2` VARCHAR(50), `Department` VARCHAR(50), `Desgnition` VARCHAR(50), `Status` VARCHAR(50), `Qualifiacations` VARCHAR(50), `Gender` VARCHAR(50), `Datehired` DATE, `Image` LONGBLOB, `PostCode` INT, `Apartment` VARCHAR(50), `Salary` INT, `MobileNO` BIGINT, PRIMARY KEY (`Emp_ID`));
